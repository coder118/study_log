
    class Solution {
        public int solution(int[] numbers) {
            int answer = 0;
            //문제 정의
            //배열이 주어짐, 배열 내부 숫자값을 곱해서 최대값을 구해라.
            //배열 내부의 값을 다루는 문제.
            
            // 1. 배열 내부의 값을 확인해야 한다
            
            //2. 확인한 배열 값들의 곱에서 최대값을 찾아야 한다.
            //문제 2-1. 곱은 2개의 값이 있어야 한다. 근데 반복문의 특성상 지난값은 저장이 되지 않는다.
            //=> 첫번째 값을 저장하는 반복문과 두번째값을 순회하는 반복문을 사용해서 해결 가능해 보임.
            
            int max = 0;
            int temp = 0;
            for(int first = 0; first<numbers.length; first++){
                for(int sec =first+1; sec<numbers.length; sec++){
                    temp = numbers[first]*numbers[sec];
                    if(temp>max){
                        max = temp;
                    }
                }
            }
            
            answer = max;
            
            return answer;
        }
    }
